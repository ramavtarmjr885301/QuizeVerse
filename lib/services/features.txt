Future<void> bulkUploadQuestionsBatch1() async {
  final firestore = FirebaseFirestore.instance;

  // Duplicate-prevention: check if this exact batch was already uploaded
  final marker = await firestore.collection('_migrations').doc('batch1_uploaded').get();
  if (marker.exists) {
    debugPrint('⚠️ Batch 1 already uploaded, skipping.');
    return;
  }

  final batch = firestore.batch();
  final collection = firestore.collection('questions');

  for (final q in _batch1Questions) {
    final docRef = collection.doc();
    batch.set(docRef, q);
  }

  batch.set(firestore.collection('_migrations').doc('batch1_uploaded'), {
    'uploadedAt': FieldValue.serverTimestamp(),
    'count': _batch1Questions.length,
  });

  try {
    await batch.commit();
    debugPrint('✅ Uploaded ${_batch1Questions.length} questions (Batch 1)');
  } catch (e) {
    debugPrint('❌ Bulk upload failed: $e');
  }
}
